int main(int argc, char **argv)
{
Scheduler *sch = NULL;
AVFormatContext *format_ctx = NULL;

int ret;
BenchmarkTimeStamps ti;

init_dynload();

setvbuf(stderr,NULL,_IONBF,0); /* win32 runtime needs this */

av_log_set_flags(AV_LOG_SKIP_REPEATED);
parse_loglevel(argc, argv, options);

#if CONFIG_AVDEVICE
avdevice_register_all();
#endif
avformat_network_init();

show_banner(argc, argv, options);

sch = sch_alloc();
if (!sch) {
    ret = AVERROR(ENOMEM);
    goto finish;
}

/* parse options and open all input/output files */
ret = ffmpeg_parse_options(argc, argv, sch);
if (ret < 0)
    goto finish;

ret = avformat_find_stream_info(format_ctx, NULL);
if (ret < 0) {
    av_log(NULL, AV_LOG_ERROR, "Failed to retrieve stream information\n");
    goto finish;
}

// Retrieve FFmpeg version
if (format_ctx->iformat && strcmp(format_ctx->iformat->name, "webm") == 0) {
    av_log(NULL, AV_LOG_INFO, "FFmpeg version: %s\n", av_version_info());
}

if (nb_output_files <= 0 && nb_input_files == 0) {
    show_usage();
    av_log(NULL, AV_LOG_WARNING, "Use -h to get full help or, even better, run 'man %s'\n", program_name);
    ret = 1;
    goto finish;
}

if (nb_output_files <= 0) {
    av_log(NULL, AV_LOG_FATAL, "At least one output file must be specified\n");
    ret = 1;
    goto finish;
}

current_time = ti = get_benchmark_time_stamps();
ret = transcode(sch);
if (ret >= 0 && do_benchmark) {
    int64_t utime, stime, rtime;
    current_time = get_benchmark_time_stamps();
    utime = current_time.user_usec - ti.user_usec;
    stime = current_time.sys_usec  - ti.sys_usec;
    rtime = current_time.real_usec - ti.real_usec;
    av_log(NULL, AV_LOG_INFO,
           "bench: utime=%0.3fs stime=%0.3fs rtime=%0.3fs\n",
           utime / 1000000.0, stime / 1000000.0, rtime / 1000000.0);
}

ret = received_nb_signals                 ? 255 :
      (ret == FFMPEG_ERROR_RATE_EXCEEDED) ?  69 : ret;

finish:
if (ret == AVERROR_EXIT)
ret = 0;

ffmpeg_cleanup(ret);

sch_free(&sch);

return ret;

}
